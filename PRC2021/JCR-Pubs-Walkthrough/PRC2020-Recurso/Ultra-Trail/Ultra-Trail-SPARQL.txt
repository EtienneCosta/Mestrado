    ?s ?p :Atleta;
       :Nome ?Nome.
}

3)  O top ten das equipes com mais atletas.

select (?Eq as ?ID) ?Equipa (count (?Equipa) as ?Total)  where {
    ?s ?p :Atleta;
       :pertence ?Eq.
    ?Eq :Nome ?Equipa.    	
}

group by ?Eq ?Equipa 
order by desc (?Total)
limit 10
